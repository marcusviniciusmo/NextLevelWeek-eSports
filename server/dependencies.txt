// Create a Node Project
npm init -y

// Intall Express dependency
npm install express

// Install TypeScript
npm i typescript

// Install TypeScript in development environment
npm i typescript -D

// Create a config TS file
npx tsc --init

// Added @types to Express
npm install @types/express -D

// Install ts-node-dev
npm i ts-node-dev -D

// Install Prisma
npm i prisma -D

// Prisma dependency with SQLite Database
npx prisma init --datasource-provider SQLite
